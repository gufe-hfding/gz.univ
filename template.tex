\section{分数阶导数与积分的定义}



首先, 我们将回顾函数分数阶微积分的定义. 专著\cite{Kilbas}中有更为详细的介绍.

\begin{defn}\upcite{Kilbas}(Riemann-Liouville型分数积分)
函数$f:[0,\infty)\rightarrow \mathbb{R}$以$0$为积分下限的$\alpha$ ($\alpha>0$)阶积分为
\[
I_{0^+}^{\alpha}
 f(t)=\frac{1}{\Gamma(\alpha)}\int_{0}^{t}\frac{f(s)}{(t-s)^{1-\alpha}}ds,~
  t>0,
\]
其中$\Gamma(\cdot)$是Gamma函数~~(Gamma函数:$\Gamma(s)=\int^{\infty}_{0}e^{-x}x^{s-1}dx$, $s>0$).
\end{defn}

\begin{defn}\upcite{Kilbas}(Riemann-Liouville型分数导数)
函数$f:[0,\infty)\rightarrow \mathbb{R}$以$0$为积分下限的$\alpha$($\alpha>0$)阶Riemann-Liouville导数为
\[
^LD_{0^+}^{\alpha}
  f(t)=\frac{1}{\Gamma(n-\alpha)}\frac{d^n}{dt^n}\int_{0}^{t}\frac{f(s)}{(t-s)^{\alpha+1-n}}ds,
~n-1<\alpha<n,
\]
其中, $n=[\alpha]+1$ ($[\alpha]$表示$\alpha$的整数部分).
\end{defn}

\begin{defn}\upcite{Kilbas}(经典Caputo型分数导数)
函数$f:[0,\infty)\rightarrow \mathbb{R}$以$0$为积分下限的$\alpha$($\alpha>0$)阶Caputo导数为
\[
^{C}D_{0^+}^{\alpha}
  f(t)=\frac{1}{\Gamma(n-\alpha)}\int_{0}^{t}\frac{f^{(n)}(s)}{(t-s)^{\alpha+1-n}}ds,
~n-1<\alpha<n,
\]
其中, $n=[\alpha]+1$ ($[\alpha]$表示$\alpha$的整数部分).
\end{defn}
显然, 任意常数的Caputo导数为0.

值得注意的是, 阶数为$\alpha$的Riemann-Liouville分数导数与Caputo分数导数存在着一定联系, 即为:
\begin{defn}\upcite{Kilbas}(弱Caputo型导数)
函数$f:[0,\infty)\rightarrow \mathbb{R}$以$0$为积分下限的$\alpha$($\alpha>0$)阶导数为
\begin{eqnarray*}
^{C}D_{0^+}^{\alpha}f(t)=~^LD_{0^+}^{\alpha}\bigg [
f(t)-\sum^{n-1}_{k=0}\frac {t^k}{k!} f^{(k)}(0)\bigg ],
  ~t>0,~n-1<\alpha <n,
\end{eqnarray*}
其中, $n=[\alpha]+1$ ($[\alpha]$表示$\alpha$的整数部分).
\end{defn}

\begin{lemma}\upcite{Kilbas}\label{lem-ID}
当$\alpha>0$, $n=[\alpha]+1$ $(n \notin \mathbb{N})$ 或 $n=\alpha$
$(n \in \mathbb{N})$, 如果$y(x) \in C^{(n)}[a,b]$, 则
\begin{eqnarray*}
(I^{\alpha}_{a^+}~^{C}D^{\alpha}_{a^+} y) (x)=y(x)-\sum^{n-1}_{k=0}\frac{y^{(k)}(a)}{k!}(x-a)^k.
\end{eqnarray*}
特别地, 当$0<\alpha <1$, $y(x) \in C[a,b]$, 则
$$(I^{\alpha}_{a^+}~^{C}D^{\alpha}_{a^+} y) (x)=y(x)-y(a),$$
$$(^{C}D^{\alpha}_{a^+}~I^{\alpha}_{a^+} y) (x)=y(x).$$
\end{lemma}



\section{重要引理和不动点定理}



\begin{defn}\upcite{R. Precup09}
当矩阵$M$满足$M^k\rightarrow 0$~($k \rightarrow \infty$)时, 则称$M$是收敛于零的.
\end{defn}

\begin{lemma}\upcite{R. Precup09, Precup-KD}\label{lem-1}
下述命题等价:\\
$(i)$ $M$是一个收敛于零的矩阵;
\\
$(ii)$ $I-M$是非奇异的, 并且$(I-M)^{-1}= \sum_{k=0}^{\infty}M^k$, 其中
$I$表示与$M$具有相同阶数的单位矩阵;
\\
$(iii)$ 对每个$\lambda \in \mathbb{C}$且$|M-\lambda I|=0$, 存在$|\lambda|<1$；
\\
$(iv)$ $I-M$是非奇异的, 并且逆元$(I-M)^{-1}$的元素都是非负元素.
\end{lemma}

\begin{lemma}\upcite{Nica-FIX}\label{lem-2}
如果$A$是一个收敛于零的矩阵, 并且对于同阶的矩阵$B$, 其元素足够小, 则$A+B$也是收敛于零的.
\end{lemma}

\begin{defn}\upcite{Nica-FIX,NicaP}\label{Matrix-vector}
$X$是一个非空集合, 在$X$上定义一个范数$d:X\times X\rightarrow \mathbb{R}_+^n$使其满足\\
$(i)$ 对任意$u,v\in X$, 满足$d(u,v)\geq 0$, 并且当且仅当$u=v$时$d(u,v)=0$;
\\
$(ii)$ 对任意$u,v\in X$, $d(u,v)=d(v,u)$;\\
$(iii)$ 对任意$u,v\in X$, $d(u,v)\leq d(u,w)+d(w,v)$.
\end{defn}

如果$x,y \in \mathbb{R}^n$, $x=(x_1,x_2,\cdots x_n),~
y=(y_1,y_2,\cdots y_n)$, 当$i=1,2\cdots n$, $x_i \leq y_i$ 时,
称$x\leq y$. $(X,d)$是广义的矩阵空间. 类同于一般度量空间的性质,
该空间也具有收敛性和完备性.

\begin{defn}\upcite{Nica-FIX,NicaP}\label{contractive}
如果存在一个收敛于零的矩阵$M$满足
\[
d(T(u),T(v))\leq M d(u,v),~~~~~u,v \in X,
\]
称算子$T:X\rightarrow X$在$X$上基于范数$d$是收敛的.
\end{defn}

\begin{lemma}(Arzala-Ascoli定理) 设$S=\{s(t)\}$是Banach空间$X$的一些连续映射所构成的函数族, 满足: $(i)$ $S$一致有界, 即存在$M >0$, 使对$\forall s \in S$, $t \in J$, 有$|s(t)|\leq M;$ $(ii)$ $S$等度连续, 即对任意$\varepsilon > 0$, 存在$\delta = \delta(\varepsilon)>0$, 使对$\forall s \in S$, $t_1,t_2 \in J$, 只要$|t_1-t_2|<\delta$, 就存在$|x(t_1)-x(t_2)|<\varepsilon$. 则从$S$中必可选取一个在$J$上一致收敛的函数序列$\{s_n(t)\}$~$n=1,2,\cdots$, 且收敛于某个$s(t)$.
\end{lemma}

\begin{defn}(全连续算子)
令$X$是一个Banach空间, 如果对$X$的每一个子集$S$, $T(S)$是紧的, 那么,
算子$T:X\rightarrow X$叫做完全有界的. 更进一步,
如果$T$在$X$上是连续并且完全有界, 那么, $T$就叫$X$的全连续算子.
\end{defn}


\begin{theorem}\upcite{Agarwal-C}(Perov不动点定理)\label{perov}
设$(X,d)$是非空的完备度量空间, $T: X\rightarrow
X$是一个具有Lipschitz矩阵$M$的压缩映射,
则$T$在$X$上有唯一的不动点$u^*$, 并且对每个$u_0 \in X$,
有$d(T^{k}(u_0),u^*)\leq M^k(I-M)^{-1}d(u_0,T(u_0))$, $k \in
\mathbb{N}$.
\end{theorem}

\begin{theorem}\upcite{Agarwal-C}(Schauder不动点定理)\label{schauder}
设$X$是实Banach空间, $D$是$X$中的非空有界闭凸集,
如果算子$T:D\rightarrow D$全连续, 则算子$T$在$D$中至少有一个不动点.
\end{theorem}

\begin{theorem}\upcite{Agarwal-C}(Leray-Schauder不动点定理)\label{Leray-schauder}
设$(X,\|\cdot\|_X)$是Banach空间, $R>0$,
$T:\bar{B}_{R}(0;X)\rightarrow X$是一全连续算子, 如果对$u=\lambda
T(U)$($\lambda\in (0,1)$)的每一个解$u$满足$\|u\|<R$,
则$T$至少有一个不动点.
\end{theorem}



\section{常用积分不等式}



\begin{lemma}(H\"{o}lder不等式) 设$p,q> 1$满足$\frac{1}{p}+\frac{1}{q}=1$. 如果$u \in L^p(J,\mathbb{R})$, $v \in L^q(J,\mathbb{R})$, 那么$uv \in L^{1}(J,\mathbb{R})$ 且$$\|uv\|_{L^1}\leq\|u\|_{L^p}\|v\|_{L^q}.$$
\end{lemma}


\begin{lemma}
若$A>B>0$, 则
$$A^{\alpha}-B^{\alpha}\leq (A-B)^\alpha,~\mbox{当}~0<\alpha\leq1.$$
\end{lemma}


\begin{lemma}\label{int-lemma}
当 $0<\alpha<1$, $\tau >0$时, 对积分
\begin{eqnarray*}
Z:=\int^t_0(t-s)^{\alpha-1}e^{\tau s^\alpha}ds,~t\in [0,1].
\end{eqnarray*}
成立下述结论:
\begin{eqnarray*}
Z&\leq& \frac{e^{\tau t^\alpha}}{\alpha([\tau]+1)^{\alpha}},~~~~~[\tau]=0,\\
Z&\leq& \frac{2e^{\tau t^\alpha}}{\alpha[\tau]^{\alpha}},~~~~~~~~~~~~~~~[\tau]\geq1.
\end{eqnarray*}
\end{lemma}
\begin{proof} 取
$$Z_1:=\sum_{i=1}^{[\tau]}\int^{\frac{it}{[\tau]+1}}_{\frac{(i-1)t}{[\tau]+1}}(t-s)^{\alpha-1}e^{\tau
s^\alpha}ds,$$
$$Z_2:=\int^{t}_{\frac{[\tau]t}{[\tau]+1}}(t-s)^{\alpha-1}e^{\tau
s^\alpha}ds.$$

第一种情形: 当 $[\tau]=0$ 时, $Z=Z_2$成立.

显然,
\begin{eqnarray*}
Z_2&:=&\int^{t}_{\frac{[\tau]t}{[\tau]+1}}(t-s)^{\alpha-1}e^{\tau
s^\alpha}ds\\ &\leq&e^{\tau
t^\alpha}\int^{t}_{\frac{[\tau]t}{[\tau]+1}}(t-s)^{\alpha-1}ds
~=~\frac{t^\alpha e^{\tau t^\alpha}}{\alpha([\tau]+1)^{\alpha}}
~\leq~ \frac{e^{\tau t^\alpha}}{\alpha([\tau]+1)^{\alpha}}.
\end{eqnarray*}

第二种情形: 当$[\tau]\geq 1$时, 有$Z=Z_1+Z_2$,
$$Z_2\leq \frac{e^{\tau t^\alpha}}{\alpha[\tau]^{\alpha}}.$$

其中,
\begin{eqnarray*}
&&\int^{\frac{it}{[\tau]+1}}_{\frac{(i-1)t}{[\tau]+1}}(t-s)^{\alpha-1}e^{\tau
s^\alpha}ds\\
&\leq&\int^{\frac{it}{[\tau]+1}}_{\frac{(i-1)t}{[\tau]+1}}\left(\frac{([\tau]+1)
s}{i}-s\right)^{\alpha-1}e^{\tau
s^\alpha}ds~~\left(s\leq\frac{it}{[\tau]+1}\Longrightarrow t \geq\frac{([\tau]+1) s}{i}\right)\\
&\leq&\left(\frac{[\tau]+1}{i}-1\right)^{\alpha-1}\int^{\frac{it}{[\tau]+1}}_{\frac{(i-1)t}{[\tau]+1}}
s^{\alpha-1}e^{\tau s^\alpha}ds\\
&\leq&\left(\frac{[\tau]+1}{i}-1\right)^{\alpha-1}\int^{\frac{it}{[\tau]+1}}_{\frac{(i-1)t}{[\tau]+1}}
e^{\tau s^\alpha}d\left(\frac{1}{\alpha}s^{\alpha}\right)\\
&=&\frac{1}{\tau
\alpha}\left(\frac{[\tau]+1}{i}-1\right)^{\alpha-1}\left[e^{\tau
\left(\frac{it}{[\tau]+1}\right)^\alpha}-e^{\tau
\left(\frac{(i-1)t}{[\tau]+1}\right)^\alpha}\right],
\end{eqnarray*}
因此,
\begin{eqnarray*}
Z_1&:=&\sum_{i=1}^{[\tau]}\int^{\frac{it}{[\tau]+1}}_{\frac{(i-1)t}{[\tau]+1}}(t-s)^{\alpha-1}e^{\tau
s^\alpha}ds\\
&\leq&
\sum_{i=1}^{[\tau]}\frac{1}{\tau\alpha}\left(\frac{[\tau]+1}{i}-1\right)^{\alpha-1}\left[e^{\tau
\left(\frac{it}{[\tau]+1}\right)^\alpha}-e^{\tau\left(\frac{(i-1)t}{[\tau]+1}\right)^\alpha}\right]\\
&=&\frac{1}{\tau\alpha}\sum_{i=1}^{[\tau]}\left(\frac{[\tau]+1}{i}-1\right)^{\alpha-1}\left[e^{\tau\left(\frac{it}{[\tau]+1}\right)^\alpha}-e^{\tau\left(\frac{(i-1)t}{[\tau]+1}\right)^\alpha}\right]\\
&=&\frac{1}{\tau\alpha}\bigg\{[\tau]^{\alpha-1}\left[e^{\tau\left(\frac{t}{[\tau]+1}\right)^{\alpha}}-1\right]+\left(\frac{[\tau]+1}{2}-1\right)^{\alpha-1}\left[e^{\tau
\left(\frac{2t}{[\tau]+1}\right)^\alpha}-e^{\tau \left(\frac{t}{[\tau]+1}\right)^\alpha}\right]\\
&& +\cdots
+\left(\frac{[\tau]+1}{[\tau]}-1\right)^{\alpha-1}\left[e^{\tau
\left(\frac{[\tau]t}{[\tau]+1}\right)^\alpha}-e^{\tau
\left(\frac{([\tau]-1)t}{[\tau]+1}\right)^\alpha}\right] \bigg\}\\
&\leq&\frac{1}{\tau
\alpha}\left[\left(\frac{1}{[\tau]}\right)^{\alpha-1}e^{\tau
\left(\frac{[\tau]t}{[\tau]+1}\right)^\alpha}-[\tau]^{\alpha-1}\right]\\
&\leq&\frac{1}{\tau
\alpha}\left(\frac{1}{[\tau]}\right)^{\alpha-1}e^{\tau
\left(\frac{[\tau]t}{[\tau]+1}\right)^\alpha}\\
&\leq&\frac{1}{\tau
\alpha}\left(\frac{1}{\tau}\right)^{\alpha-1}e^{\tau
t^\alpha}\\
&\leq&\frac{e^{\tau t^\alpha}}{\alpha\tau^{\alpha}}\\
&\leq&\frac{e^{\tau t^\alpha}}{\alpha[\tau]^{\alpha}}.
\end{eqnarray*}

结合$Z_1$和$Z_2$的结论, 引理结论得到了验证.
\end{proof}



\section{解的存在唯一性定理}



\begin{theorem}\label{th-p-1}
若假设$[H1]$, $[H2]$, $[H3]$成立, 则系统 (\ref{sy.1})
有唯一解.
\end{theorem}
\begin{proof}
根据假设$[H2]$和引理\ref{lem-1}(iv), 可知, $I-M_{\theta}$是可逆的,
并且, 其逆元$(I-M_{\theta})^{-1}$中的元素都是非负元素.

定义
\[
\tilde{U}=\{(x,y)\in C[0,1]\times C[0,1]:\|x\|_C \leq \tilde{R}_1,~\|y\|_C \leq
\tilde{R}_2\},
\]
其中,
\begin{eqnarray*}
\left[
\begin{array}{l}
\tilde{R}_1\\
\tilde{R}_2
\end{array}
\right] \geq (I-M_{\theta})^{-1} \left[
\begin{array}{l}
 \tilde{M}_1\\
\tilde{M}_2
\end{array}
 \right],
\end{eqnarray*}
\[
\tilde{M}_1=\left(A_1l+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)}\right)
f_{\max},~~~\tilde{M}_2=
\left(A_2l+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)}\right) g_{\max},
\]
记, $f_{\max}=\max_{t\in [0,1]}\left|f(t,0,0)\right|$,
$g_{\max}=\max_{t\in [0,1]}\left|g(t,0,0)\right|$.


第一步: 证明对于任意的$(x,y) \in \tilde{U}$, $T(\tilde{U}) \subset
\tilde{U}$.

对任意$(x,y)\in \tilde{U}$, 定义 $M_1=\sup_{(t,x,y)\in [0,1]\times
\tilde{U}}\left|f(t,x,y)\right|$和$M_2=\sup_{(t,x,y)\in [0,1]\times
\tilde{U}}\linebreak[4] \left|g(t,x,y)\right|$.

首先, 当$0\leq t_\alpha<t_\beta\leq t_m$, 依据$[H1]$, $[H2]$, $[H3]$, 可得
\begin{eqnarray*}
&&\left|T_1(x,y)(t_\beta)-T_1(x,y)(t_\alpha)\right|\\
&=&
\bigg|-a\sum^{m}_{k=1}a_k\int_{0}^{t_k}
\frac{(t_k-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds
+\int_{0}^{t_\beta}\frac{(t_\beta-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds\\
&&+a\sum^{m}_{k=1}a_k\int_{0}^{t_k}
\frac{(t_k-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds
-\int_{0}^{t_\alpha}\frac{(t_\alpha-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds\bigg|\\
&\leq&\frac{M_1}{\Gamma(\gamma)}\left[\int^{t_\alpha}_{0}\left|(t_\beta-s)^{\gamma-1}
-(t_\alpha-s)^{\gamma-1}\right|ds
+\int^{t_\beta}_{t_\alpha}(t_\beta-s)^{\gamma-1}ds\right]\\
&\leq&\frac{M_1}{\Gamma(\gamma)}\left[\frac{1}{\gamma}
\left((t_\beta-t_\alpha)^\gamma+(t_{\beta}^\gamma-t_{\alpha}^\gamma)+(t_\beta-t_\alpha)^\gamma\right)\right]\\
&\leq&\frac{3M_1}{\Gamma(\gamma+1)}(t_\beta-t_\alpha)^\gamma,
\end{eqnarray*}
当$t_\beta\rightarrow t_\alpha$时, 可以得出
\[
\left|T_1(x,y)(t_\beta)-T_1(x,y)(t_\alpha)\right|\rightarrow
0.
\]

其次, 当$t_m<t_\alpha<t_\beta \leq 1$, 得
\begin{eqnarray*}
&&\left|T_1(x,y)(t_\beta)-T_1(x,y)(t_\alpha)\right|\\
&=& \bigg|-a\sum^{m}_{k=1}a_k\int_{0}^{t_k}
\frac{(t_k-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds
+\int_{0}^{t_m}\frac{(t_\beta-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds\\
&&+a\sum^{m}_{k=1}a_k\int_{0}^{t_k}\frac{(t_k-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds
-\int_{0}^{t_m}\frac{(t_\alpha-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds\\
&&+\int^{t_\beta}_{t_m}\frac{(t_\beta-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds
-\int^{t_\alpha}_{t_m}\frac{(t_\alpha-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds\bigg|\\
&\leq&
\int^{t_m}_{0}\frac{(t_\beta-s)^{\gamma-1}-(t_\alpha-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))\right|ds\\
&&+\int^{t_\alpha}_{t_m}\frac{(t_\beta-s)^{\gamma-1}-(t_\alpha-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))\right|ds\\
&&+\int^{t_\beta}_{t_\alpha}\frac{(t_\beta-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))\right|ds\\
&\leq& \frac{M_1}{\Gamma(\gamma)}\bigg[
\int^{t_m}_{0}\left|(t_\beta-s)^{\gamma-1}-(t_\alpha-s)^{\gamma-1}\right|ds
+\int^{t_\alpha}_{t_m}\left|(t_\beta-s)^{\gamma-1}-(t_\alpha-s)^{\gamma-1}\right|ds\\
&&+\int^{t_\beta}_{t_\alpha}(t_\beta-s)^{\gamma-1}ds\bigg]\\
&\leq&\frac{5M_1}{\Gamma(\gamma+1)}(t_\beta-t_\alpha)^\gamma,
\end{eqnarray*}
当$t_\beta\rightarrow t_\alpha$时, 显然
$\left|T_1(x,y)(t_\beta)-T_1(x,y)(t_\alpha)\right|\rightarrow
0$也成立.

因此, $T_1(\tilde{U})\subset C[0,1]\times C[0,1]$. 类似地, 可证明: $T_2(\tilde{U})\subset C[0,1]\times C[0,1]$.

下证$T(\tilde{U}) \subset \tilde{U}$.

对任意$t\in[0,t_m]$, 有
\begin{eqnarray*}
&&\left|T_1(x,y)(t)\right|\nonumber\\
&\leq&
|a|\sum^{m}_{k=1}|a_k|\int^{t_m}_{0}\frac{(t_m-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))\right|ds
+\int^{t_m}_{0}\frac{(t_m-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))\right|ds\nonumber\\
&\leq&\frac{A_1}{\Gamma(\gamma)}\int_{0}^{t_m}(t_m-s)^{\gamma-1}\left|f(s,x(s),y(s))-f(s,0,0)+f(s,0,0)\right|ds\nonumber\\
&\leq&\frac{A_1}{\Gamma(\gamma)}\int_{0}^{t_m}(t_m-s)^{\gamma-1}\left[b_1|x(s)|_{C[0,t_m]}+\tilde{b}_1|y(s)|_{C[0,t_m]}+f_{\max}\right]ds\nonumber\\
&\leq&b_1lA_1\|x\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y\|_{C[0,t_m]}+A_1lf_{\max},
\end{eqnarray*}
因此,
\begin{eqnarray}\label{P1-T1-s1-1}
\|T_1(x,y)\|_{C[0,t_m]}&\leq&b_1lA_1\|x\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y\|_{C[0,t_m]}+A_1lf_{\max}.
\end{eqnarray}

对任意$t\in[t_m,1]$, 有
\begin{eqnarray*}
&&\left|T_1(x,y)(t)\right|\nonumber\\
&\leq&b_1lA_1\|x\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y\|_{C[0,t_m]}+A_1lf_{\max}
+\int^{t}_{t_m}\frac{(t-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))\right|ds\nonumber\\
&\leq&b_1lA_1\|x\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y\|_{C[0,t_m]}+A_1lf_{\max}\nonumber\\
&&+\frac{1}{\Gamma(\gamma)}\int^{t}_{t_m}(t-s)^{\gamma-1}\left(c_1|x(s)|+\tilde{c}_1|y(s)|+f_{\max}\right)ds\nonumber\\
&\leq&b_1lA_1\|x\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y\|_{C[0,t_m]}+A_1lf_{\max}\nonumber\\
&&+\frac{c_1}{\Gamma(\gamma)}\int^{t}_{t_m}(t-s)^{\gamma-1}|x(s)|e^{-\theta(s-t_m)^\gamma}e^{\theta(s-t_m)^\gamma}ds \nonumber\\
&&+\frac{\tilde{c}_1}{\Gamma(\gamma)}\int^{t}_{t_m}(t-s)^{\gamma-1}|y(s)|e^{-\theta(s-t_m)^\gamma}
e^{\theta(s-t_m)^\gamma}ds
+\frac{1}{\Gamma(\gamma)}\int^{t}_{t_m}(t-s)^{\gamma-1}f_{\max}ds\nonumber\\
&\leq&b_1lA_1\|x\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y\|_{C[0,t_m]}+A_1lf_{\max}\nonumber\\
&&+\frac{2c_1}{\Gamma(\gamma+1) [\theta]^\gamma}e^{\theta
(t-t_m)^\gamma}\|x\|_{C[t_m,1]}+\frac{2\tilde{c}_1}{\Gamma(\gamma+1)
[\theta]^\gamma}e^{\theta
(t-t_m)^\gamma}\|y\|_{C[t_m,1]}+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)} f_{\max},\nonumber\\
\end{eqnarray*}
此处用到引理\ref{int-lemma}的结论, 即对任意$t \in[t_m,1]$, 有
\begin{eqnarray*}
\int^{t}_{t_m}(t-s)^{\gamma-1}e^{\theta(s-t_m)^\gamma}ds
&\leq&\frac{2e^{\theta(t-t_m)^\gamma}}{\gamma[\theta]^\gamma}.
\end{eqnarray*}

两边同时除以$e^{\theta(t-t_m)^\gamma}$, 并取最大值, 得到
\begin{eqnarray}\label{P1-T1-s1-2}
\|T_1(x,y)\|_{C[t_m,1]} &\leq& b_1lA_1\|x\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y\|_{C[0,t_m]}+\left(A_1l+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)}\right)f_{\max}\nonumber\\
&&+\frac{2c_1}{\Gamma(\gamma+1)
[\theta]^\gamma}\|x\|_{C[t_m,1]}+\frac{2\tilde{c}_1}{\Gamma(\gamma+1)
[\theta]^\gamma}\|y\|_{C[t_m,1]}.
\end{eqnarray}

结合(\ref{P1-T1-s1-1})和(\ref{P1-T1-s1-2}), 得
\begin{eqnarray}\label{P1-T1-bod}
\|T_1(x,y)\|_C\leq\left(b_1lA_1+\frac{2c_1}{\Gamma(\gamma+1)
[\theta]^\gamma}\right)\|x\|_C
+\left(\tilde{b}_1lA_1+\frac{2\tilde{c}_1}{\Gamma(\gamma+1)
[\theta]^\gamma}\right)\|y\|_C +\tilde{M}_1,\nonumber\\
\end{eqnarray}
其中,
\[
\tilde{M}_1=\max\left\{A_1lf_{\max},
\left(A_1l+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)}\right)f_{\max}\right\}=
\left(A_1l+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)}\right)f_{\max}.
\]

类似地, 可得
\begin{eqnarray}\label{P1-T2-bod}
\|T_2(x,y)\|_C \leq \left(B_1lA_2+\frac{2C_1}{\Gamma(\gamma+1)
[\theta]^\gamma}\right)\|x\|_C
+\left(\tilde{B}_1lA_2+\frac{2\tilde{C}_1}{\Gamma(\gamma+1)
[\theta]^\gamma}\right)\|y\|_C +\tilde{M}_2,\nonumber\\
\end{eqnarray}
其中,
\[
\tilde{M}_2=\max\left\{A_2lg_{\max},
\left(A_2l+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)}\right)g_{\max}\right\}=
\left(A_2l+\frac{(1-t_m)^{\gamma}}{\Gamma(\gamma+1)}\right)g_{\max}.
\]

结合(\ref{P1-T1-bod})和(\ref{P1-T2-bod}), 得到
\begin{eqnarray*}
\left[
\begin{array}{l}
\|T_1(x,y)\|_C\\
\|T_2(x,y)\|_C
\end{array}
\right] \leq  \left[
\begin{array}{l}
\tilde{R}_1\\
\tilde{R}_2
\end{array}
 \right],
\end{eqnarray*}
满足$T(\tilde{U})\subset \tilde{U}$.


第二步: 证明$T$是压缩映射.


对任意的$(x,y),(\bar{x},\bar{y}) \in \tilde{U}$, 任意$t\in[0,t_m]$, 依据$[H1]$,$[H2]$,$[H3]$, 可得,
\begin{eqnarray*}
&&\left|T_1(x,y)(t)-T_1(\bar{x},\bar{y})(t)\right|\\
&=& \bigg|-a\sum^{m}_{k=1}a_k\int_{0}^{t_k}
\frac{(t_k-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds
+\int_{0}^{t}\frac{(t-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,x(s),y(s))ds\\
&&+a\sum^{m}_{k=1}a_k\int_{0}^{t_k}
\frac{(t_k-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,\bar{x}(s),\bar{y}(s))ds
-\int_{0}^{t}\frac{(t-s)^{\gamma-1}}{\Gamma(\gamma)}f(s,\bar{x}(s),\bar{y}(s))ds\bigg|\\
&\leq&
|a|\sum^{m}_{k=1}|a_k|\int^{t_m}_{0}\frac{(t_m-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))
-f(s,\bar{x}(s),\bar{y}(s))\right|ds\\
&&+\int^{t_m}_{0}\frac{(t_m-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))
-f(s,\bar{x}(s),\bar{y}(s))\right|ds\\
&\leq&\frac{A_1}{\Gamma(\gamma)}\int^{t_m}_{0}(t_m-s)^{\gamma-1}\left[b_1|x(s)-\bar{x}(s)|_{C[0,t_m]}+\tilde{b}_1|y(s)-\bar{y}(s)|_{C[0,t_m]}\right]ds\\
&\leq&b_1lA_1\|x-\bar{x}\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y-\bar{y}\|_{C[0,t_m]},
\end{eqnarray*}
因此,
\begin{eqnarray}\label{P1-T1-s2-1}
\|T_1(x,y)-T_1(\bar{x},\bar{y})\|_{C[0,t_m]}\leq
b_1lA_1\|x-\bar{x}\|_{C[0,t_m]}
+\tilde{b}_1lA_1\|y-\bar{y}\|_{C[0,t_m]}.
\end{eqnarray}

对任意$t \in [t_m,1]$, 任意$\theta>0$, 可得
\begin{eqnarray*}
&&\left|T_1(x,y)(t)-T_1(\bar{x},\bar{y})(t)\right|\\
&\leq&b_1lA_1\|x-\bar{x}\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y-\bar{y}\|_{C[0,t_m]}\\
&&+\int^{t}_{t_m}\frac{(t-s)^{\gamma-1}}{\Gamma(\gamma)}\left|f(s,x(s),y(s))
-f(s,\bar{x}(s),\bar{y}(s))\right|ds,
\end{eqnarray*}
注意到
\begin{eqnarray*}
&&\int^{t}_{t_m}(t-s)^{\gamma-1}\left|f(s,x(s),y(s))-f(s,\bar{x}(s),\bar{y}(s))\right|ds\\
&\leq&c_1\int^{t}_{t_m}(t-s)^{\gamma-1}\left|x(s)-\bar{x}(s)\right|e^{-\theta(s-t_m)^\gamma}e^{\theta(s-t_m)^\gamma}ds\\
&&+\tilde{c}_1\int^{t}_{t_m}(t-s)^{\gamma-1}\left|y(s)-\bar{y}(s)\right|e^{-\theta(s-t_m)^\gamma}e^{\theta(s-t_m)^\gamma}ds\\
&\leq&\frac{2c_1e^{\theta(t-t_m)^\gamma}}{\gamma[\theta]^\gamma}\|x-\bar{x}\|_{C[t_m,1]}
+\frac{2\tilde{c}_1e^{\theta(t-t_m)^\gamma}}{\gamma[\theta]^\gamma}\|y-\bar{y}\|_{C[t_m,1]},
\end{eqnarray*}
此处用到引理\ref{int-lemma}的结论, 即对任意$t \in[t_m,1]$, 有
\begin{eqnarray*}
\int^{t}_{t_m}(t-s)^{\gamma-1}e^{\theta(s-t_m)^\gamma}ds
&\leq&\frac{2e^{\theta(t-t_m)^\gamma}}{\gamma[\theta]^\gamma}.
\end{eqnarray*}


因此,
\begin{eqnarray*}
&&\left|T_1(x,y)(t)-T_1(\bar{x},\bar{y})(t)\right|\\
&\leq&b_1lA_1\|x-\bar{x}\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y-\bar{y}\|_{C[0,t_m]}\\
&&+\frac{2c_1}{\Gamma(\gamma+1)[\theta]^\gamma}e^{\theta(t-t_m)^\gamma}\|x-\bar{x}\|_{C[t_m,1]}
+\frac{2\tilde{c}_1}{\Gamma(\gamma+1)[\theta]^\gamma}e^{\theta(t-t_m)^\gamma}\|y-\bar{y}\|_{C[t_m,1]}.
\end{eqnarray*}
进一步,
\begin{eqnarray}\label{P1-T1-s2-2}
&&\|T_1(x,y)-T_1(\bar{x},\bar{y})\|_{C[t_m,1]}\nonumber\\
&\leq&b_1lA_1\|x-\bar{x}\|_{C[0,t_m]}+\tilde{b}_1lA_1\|y-\bar{y}\|_{C[0,t_m]}\nonumber\\
&&+\frac{2c_1}{\Gamma(\gamma+1)[\theta]^\gamma}\|x-\bar{x}\|_{C[t_m,1]}+\frac{2\tilde{c}_1}{\Gamma(\gamma+1)[\theta]^\gamma}\|y-\bar{y}\|_{C[t_m,1]}.
\end{eqnarray}
结合不等式(\ref{P1-T1-s2-1})和(\ref{P1-T1-s2-2}), 得到,
\begin{eqnarray}\label{P1-T1-con}
&&\|T_1(x,y)-T_1(\bar{x},\bar{y})\|_C\nonumber\\
&\leq&\left(b_1lA_1+\frac{2c_1}{\Gamma(\gamma+1)[\theta]^\gamma}\right)\|x-\bar{x}\|_C
+\left(\tilde{b}_1lA_1+\frac{2\tilde{c}_1}{\Gamma(\gamma+1)[\theta]^\gamma}\right)\|y-\bar{y}\|_C.
\end{eqnarray}

类似地,
\begin{eqnarray}\label{P1-T2-con}
&&\|T_2(x,y)-T_2(\bar{x},\bar{y})\|_C\nonumber\\
&\leq&\left(B_1lA_2+\frac{2C_1}{\Gamma(\gamma+1)[\theta]^\gamma}\right)\|x-\bar{x}\|_C
+\left(\tilde{B}_1lA_2+\frac{2\tilde{C}_1}{\Gamma(\gamma+1)[\theta]^\gamma}\right)\|y-\bar{y}\|_C.\nonumber\\
\end{eqnarray}

由不等式(\ref{P1-T1-con}), (\ref{P1-T2-con}), 可得
\[
\|T(u)-T(\bar{u})\|_C\leq M_\theta\|u-\bar{u}\|_C,
\]
其中,
\begin{eqnarray*}
M_\theta=  \left[
\begin{array}{l}
b_1 l A_1+\frac{2c_1}{\Gamma(\gamma+1)[\theta]^\gamma} ~~~~~\tilde{b}_1 l A_1+\frac{2\tilde{c}_1}{\Gamma(\gamma+1)[\theta]^\gamma}\\
B_1 l A_2+\frac{2C_1}{\Gamma(\gamma+1)[\theta]^\gamma} ~~~\tilde{B}_1 l
A_2+\frac{2\tilde{C}_1}{\Gamma(\gamma+1)[\theta]^\gamma}
\end{array}
\right].
\end{eqnarray*}

根据定理\ref{perov},
可知系统(\ref{sy.1})在$\tilde{U}$上有唯一的不动点.
这也就意味着系统(\ref{sy.1})在$\tilde{U}$上有唯一解.
\end{proof}


为了验证定理\ref{th-p-1}, 考虑如下例子:
\begin{example}
考虑分数阶系统:
\begin{equation}\label{exam-p-1}
\left\{
\begin{array}{l}
^{C}D_{0,t}^{\frac{2}{3}}x(t)=\frac{1}{5}\frac{y^2(t)}{1+y^2(t)}\sin(2x(t))+1=:f(t,x(t),y(t)),~t\in [0,1],\\
^{C}D_{0,t}^{\frac{2}{3}}y(t)=\frac{1}{4}\frac{y^2(t)}{1+y^2(t)}\cos(2x(t))+2=:g(t,x(t),y(t)),\\
x(0)+2x(\frac{1}{8})=0,\\
y(0)+3y(\frac{1}{8})=0.
\end{array}
\right.
\end{equation}

显然,
\begin{eqnarray*}
\sup_{x,y \in \mathbb{R}}\left|\frac{\partial f(x,y)}{\partial
x}\right|\leq \frac{2}{5}:=b_1, ~~\sup_{x,y \in
\mathbb{R}}\left|\frac{\partial f(x,y)}{\partial y}\right|\leq
\frac{3\sqrt{3}}{40}:=\tilde{b}_1,\\
\sup_{x,y \in \mathbb{R}}\left|\frac{\partial g(x,y)}{\partial
x}\right|\leq \frac{1}{2}:=B_1,  ~~\sup_{x,y \in
\mathbb{R}}\left|\frac{\partial g(x,y)}{\partial y}\right|\leq
\frac{3\sqrt{3}}{32}:=\tilde{B}_1,
\end{eqnarray*}
从而有
\begin{eqnarray*}
M_{0}
 &=&\left[
\begin{array}{l}
\frac{1}{6\Gamma(5/3)}
~~~~\frac{\sqrt{3}}{32\Gamma(5/3)}\\
\frac{7}{32\Gamma(5/3)} ~~~\frac{21\sqrt{3}}{512\Gamma(5/3)}
\end{array}
\right],
\end{eqnarray*}
关于矩阵$M_{0}$, 我们知道其特征值$\lambda_1=0$和$\lambda_2=0.2633$.
因此, 根据引理\ref{lem-1}可知$M_{0}$是收敛于零的, 同时, 依据引理\ref{lem-2}当取足够大的$\theta>0$时, $M_\theta$ 收敛于零也是成立的.

至此, 定理\ref{th-p-1}中的假设条件都满足,
则系统(\ref{exam-p-1})有唯一解.
\end{example}

\chapter{中国科学院研究生院学位论文撰写要求}
\label{chap:requires}

\textbf{国科大的学位论文要求附最后，仅供参考！}

学位论文是为申请学位而撰写的学术论文，是评判学位申请者学术水平的主要依据，
也是学位申请者获得学位的必要条件之一。为规范和统一我院研究生学位论文的写作，
根据《中华人民共和国学位条例暂行实施办法》的有关规定，提出以下要求：

\section{基本要求}

学位论文必须是一篇（或由一组论文组成的一篇）系统的、完整的学术论文。
学位论文应是学位申请者本人在导师的指导下独立完成的研究成果，
不得抄袭和剽窃他人成果。学位论文的学术观点必须明确，且逻辑严谨，文字通畅。

\subsection{硕士学位论文}

硕士学位论文要注意在基础学科或应用学科中选择有价值的课题，
对所研究的课题有新的见解，并能表明作者在本门学科上掌握了坚实的基础理论和
系统的专门知识，具有从事科学研究工作或独立担负专门技术工作的能力。

硕士学位论文工作一般在硕士生完成培养计划所规定的课程学习后开始，
应包括文献阅读、开题报告、拟定并实施工作计划、科研调查、实验研究、理论分析
和文字总结等工作环节。硕士学位论文必须有一定的工作量。在论文题目确定后，
用于论文工作的时间一般不得少于一年半。

\subsection{博士学位论文}

博士学位论文要选择在国际上属于学科前沿的课题或对国家经济建设和社会发展
有重要意义的课题，要突出论文在科学和专门技术上的创新性和先进性，
并能表明作者在本门学科上掌握了坚实宽广的基础理论和系统深入的专门知识，
具有独立从事科学研究工作的能力。

博士学位论文工作是培养博士学位研究生最重要的环节，其工作时间一般不应少于两年。
博士研究生入学后，要在导师指导下确定科研方向，收集资料，阅读文献，
进行调查研究，选择研究课题。一般在第二学期，最迟在第三学期通过开题报告
并制定论文工作计划，之后根据论文工作计划分阶段报告科研和论文工作进展情况。

\section{学位论文的组成部分和排列顺序}

学位论文一般由以下几个部分组成：封面、论文摘要、论文目录、正文、参考文献、
发表文章目录、致谢等。

\subsection{封面}

根据原国家标准局《科学技术报告、学位论文和学术论文的编写格式》
（国家标准GB7713-87）的封面要求，特规定中国科学院研究生院研究生学位论文的
封面格式（见样张1和样张2），并提出以下具体要求：

\subsubsection{分类号}

必须在封面左上角注明分类号。一般应注明《中国图书资料分类法》的类号，
同时注明《国际十进分类法UDC》的类号。

\subsubsection{编号}

各培养单位自定。

\subsubsection{密级}

论文必须按国家规定的保密条例在右上角注明密级（公开型论文可不注明密级）。

\subsubsection{论文题目}

学位论文题目应当简明扼要地概括和反映出论文的核心内容，一般不宜超过20个字，
必要时可加副标题。

\subsubsection{指导教师}

指导教师必须是被批准上岗的指导教师。

\subsubsection{申请学位级别}

填硕士学位或博士学位。

\subsubsection{学科、专业名称}

按国家颁布的学科、专业目录中的名称填写。

\subsubsection{论文提交日期和论文答辩日期}

按实际提交和答辩日期填写。

\subsubsection{培养单位}

填写培养单位全称。

\subsubsection{学位授予单位}

填写``中国科学院研究生院''。

\subsection{论文摘要}

论文摘要应概括地反映出本论文的主要内容，主要说明本论文的研究目的、内容、
方法、成果和结论。要突出本论文的创造性成果或新见解，不要与引言相混淆。
中文摘要力求语言精炼准确，字数在500字左右。英文摘要内容要与中文摘要内容一致。
并在英文题目下面第一行写研究生姓名。专业名称用括号括起后，置于姓名之后。
研究生姓名下面的一行写导师姓名，格式为：Directed by......。
无论中英文摘要都必须在摘要页的最下方另起一行，注明本文的关键词（3～5个）。

\subsection{论文目录}

论文目录是论文的提纲，也是论文各章节组成部分的小标题。

\subsection{正文}

正文是学位论文的主体和核心部分，不同学科专业和不同的选题可以有不同的写作方式。
正文一般包括以下几个方面：

\subsubsection{引言}

引言是学位论文主体部分的开端，要求言简意赅，不要与摘要雷同或成为摘要的注解。
除了说明研究目的、方法、结果等外，还应评述国内外研究现状和相关领域中已有的研究成果；
介绍本项研究工作前提和任务，理论依据和实验基础，涉及范围和预期结果以及该论文
在已有的基础上所解决的问题。

\subsubsection{各具体章节}

\subsubsection{结论}

结论是学位论文最终和总体的结论，是整篇论文的归宿。应精炼、准确、完整。
着重阐述作者研究的创造性成果及其在本研究领域中的意义，还可进一步提出
需要讨论的问题和建议。

\subsection{参考文献}

学位论文的撰写应本着严谨求实的科学态度，凡有引用他人成果之处，
均应按论文中所引用的顺序列于文末。参考文献的著录均应符合国家有关标准
（按照GB7714-87 《文后参考文献著录格式》执行）。

1．文献是期刊时，书写格式为： 序号\ 作者. 文章题目. 期刊名,
年份(期数):起止页码

2．文献是图书时，书写格式为： 序号\ 作者. 书名. 版次. 出版地：出版单位，年份.
起止页码

\subsection{发表文章目录}

指学位申请者在学期间在各类正式刊物上发表或已被接受的学术论文。

\subsection{致谢}

表达作者对完成论文和学业提供帮助的老师、同学、领导、同事及亲属的感激之情。

\section{学位论文的书写、装订要求}

（一）中国科学院研究生院研究生学位论文必须用中文书写

1. 论文``题目''：黑体小三号

2. 论文``章''：黑体四号

3. 论文``节''：黑体小四号

4. 正文：宋体小四号

5. 为美观方便起见，要有页眉，奇数页注明每一章名称，偶数页注明论文题目。

为了便于国际合作与交流，学位论文亦可有英文或其它文字的副本。

（二）文中的图表、附注、参考文献、公式一律采用阿拉伯数字连续（或分章）编号。
如图1，表1，附注：1，文献（1），公式（1）。图序及图名置于图的下方；
表序及表名置于表的上方；论文中的公式编号用括弧括起来写在右边行末，其间不加虚线。

（三）文中所用单位一律采用国务院发布的《中华人民共和国法定计量单位》，
单位名称和符号的书写方式，应采用国际通用符号。

（四）学位论文封面采用全院统一格式，封面用纸为150克花纹纸，博士学位论文
封面颜色为红色，硕士学位论文封面颜色为蓝色（见样张博士学位论文封面、
硕士学位论文封面）。

（五）学位论文一律用A4打印纸装订。
